91  !> @param coolist_ijk a coordinate list (sparse tensor) of which index
92  !> 2 and 3 will be contracted.
93  !> @param arr_j the vector to be contracted with index 2 of coolist_ijk
94  !> @param arr_k the vector to be contracted with index 3 of coolist_ijk
95  !> @param res vector (buffer) to store the result of the contraction
96  !> @remark Note that it is NOT safe to pass `arr_j`/`arr_k` as a result buffer,
97  !> as this operation does multiple passes.
98  SUBROUTINE sparse_mul3(coolist_ijk, arr_j, arr_k, res)
99  TYPE(coolist), DIMENSION(ndim), INTENT(IN):: coolist_ijk
100  REAL(KIND=8), DIMENSION(0:ndim), INTENT(IN) :: arr_j, arr_k
101  REAL(KIND=8), DIMENSION(0:ndim), INTENT(OUT) :: res
102  INTEGER :: i,j,k,n
103  res=0.d0
104  DO i=1,ndim
105  DO n=1,coolist_ijk(i)%nelems
106  j=coolist_ijk(i)%elems(n)%j
107  k=coolist_ijk(i)%elems(n)%k
108  res(i) = res(i) + coolist_ijk(i)%elems(n)%v * arr_j(j)*arr_k(k)
109  END DO
110  END DO

113  !> Sparse multiplication of two tensors to determine the Jacobian:
114  !> \f[J_{i,j} = {\displaystyle \sum_{k=0}^{ndim}} \left( \mathcal{T}_{i,j,k} + \mathcal{T}_{i,k,j} \right) \, a_k.\f]
115  !> It's implemented slightly differently: for every \f$\mathcal{T}_{i,j,k}\f$, we add to \f$J_{i,j}\f$ as follows:
116  !> \f[J_{i,j} = J_{i,j} + \mathcal{T}_{i,j,k} \, a_k \\ J_{i,k} = J_{i,k} + \mathcal{T}_{i,j,k} \, a_j\f]
117  !> This version return a coolist (sparse tensor).
118  !> @param coolist_ijk a coordinate list (sparse tensor) of which index
119  !> 2 or 3 will be contracted.

121  !> @param jcoo_ij a coolist (sparse tensor) to store the result of the contraction
122  SUBROUTINE jsparse_mul(coolist_ijk, arr_j, jcoo_ij)

124  TYPE(coolist), DIMENSION(ndim), INTENT(OUT):: jcoo_ij
125  REAL(KIND=8), DIMENSION(0:ndim), INTENT(IN) :: arr_j

204  !> Routine to simplify a coolist (sparse tensor). For each index \f$i\f$, it upper triangularize the matrix
205  !> \f[\mathcal{T}_{i,j,k} \qquad 0 \leq j,k \leq ndim.\f]
206  !> @param tensor a coordinate list (sparse tensor) which will be simplified.
207  SUBROUTINE simplify(tensor)
208  TYPE(coolist), DIMENSION(ndim), INTENT(INOUT):: tensor
209  INTEGER :: i,j,k
210  INTEGER :: li,lii,liii,n
211  DO i= 1,ndim
212  n=tensor(i)%nelems
213  DO li=n,2,-1
214  j=tensor(i)%elems(li)%j
215  k=tensor(i)%elems(li)%k
216  DO lii=li-1,1,-1
217  IF ((j==tensor(i)%elems(lii)%j).AND.(k==tensor(i)%elems(lii)%k)) THEN
218  ! Found another entry with the same i,j,k: merge both into
219  ! the one listed first (of those two).
220  tensor(i)%elems(lii)%v=tensor(i)%elems(lii)%v+tensor(i)%elems(li)%v
221  ! Shift the rest of the items one place down.
222  DO liii=li+1,n
223  tensor(i)%elems(liii-1)%j=tensor(i)%elems(liii)%j
224  tensor(i)%elems(liii-1)%k=tensor(i)%elems(liii)%k
225  tensor(i)%elems(liii-1)%v=tensor(i)%elems(liii)%v
226  END DO
227  tensor(i)%nelems=tensor(i)%nelems-1
228  ! Here we should stop because the li no longer points to the
229  ! original i,j,k element
230  EXIT
231  ENDIF
232  ENDDO
233  ENDDO
234  n=tensor(i)%nelems
235  DO li=1,n
236  ! Clear new "almost" zero entries and shift rest of the items one place down.
237  ! Make sure not to skip any entries while shifting!
238  DO WHILE (abs(tensor(i)%elems(li)%v) < real_eps)
239  DO liii=li+1,n
240  tensor(i)%elems(liii-1)%j=tensor(i)%elems(liii)%j
241  tensor(i)%elems(liii-1)%k=tensor(i)%elems(liii)%k
242  tensor(i)%elems(liii-1)%v=tensor(i)%elems(liii)%v
243  ENDDO
244  tensor(i)%nelems=tensor(i)%nelems-1
245  ENDDO
246  ENDDO
247 
248  ENDDO
249  END SUBROUTINE simplify
