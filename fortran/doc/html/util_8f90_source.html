42  !> Random generator initialization routine
43  SUBROUTINE init_random_seed()
44  USE iso_fortran_env, only: int64
45  USE ifport !, only: getpid
46  IMPLICIT NONE
47  INTEGER, ALLOCATABLE :: seed_loc(:)
48  INTEGER :: i, n, un, istat, dt(8), pid
49  INTEGER(int64) :: t
50 
51  CALL random_seed(size = n)
52  ALLOCATE(seed_loc(n))
53  ! First try IF the OS provides a random number generator
54  OPEN(newunit=un, file="/dev/urandom", access="stream", &
55  form="unformatted", action="read", status="old", iostat=istat)
56  IF (istat == 0) THEN
57  READ(un) seed_loc
58  CLOSE(un)
59  ELSE
60  ! Fallback to XOR:ing the current time and pid. The PID is
61  ! useful in case one launches multiple instances of the same
62  ! program in parallel.
63  CALL system_clock(t)
64  IF (t == 0) THEN
65  CALL date_and_time(values=dt)
66  t = (dt(1) - 1970) * 365_int64 * 24 * 60 * 60 * 1000 &
67  + dt(2) * 31_int64 * 24 * 60 * 60 * 1000 &
68  + dt(3) * 24_int64 * 60 * 60 * 1000 &
69  + dt(5) * 60 * 60 * 1000 &
70  + dt(6) * 60 * 1000 + dt(7) * 1000 &
71  + dt(8)
72  END IF
73  pid = getpid()
74  t = ieor(t, int(pid, kind(t)))
75  DO i = 1, n
76  seed_loc(i) = lcg(t)
77  END DO
78  END IF
79  CALL random_seed(put=seed_loc)
80  contains
81  ! This simple PRNG might not be good enough for real work, but is
82  ! sufficient for seeding a better PRNG.
83  FUNCTION lcg(s)
84  integer :: lcg
85  integer(int64) :: s
86  IF (s == 0) THEN
87  s = 104729
88  ELSE
89  s = mod(s, 4294967296_int64)
90  END IF
91  s = mod(s * 279470273_int64, 4294967291_int64)
92  lcg = int(mod(s, int(huge(0), int64)), kind(0))
93  END FUNCTION lcg
94  END SUBROUTINE init_random_seed
